<script>// Constants

const canvasWidth = 800;

const canvasHeight = 600;

const shipWidth = 30;

const shipHeight = 30;

const asteroidWidth = 50;

const asteroidHeight = 50;


// Game variables

let shipX = canvasWidth / 2;

let shipY = canvasHeight / 2;

let asteroids = [];


// Create canvas

const canvas = document.getElementById("gameCanvas");

canvas.width = canvasWidth;

canvas.height = canvasHeight;

const ctx = canvas.getContext("2d");


// Ship image placeholder

const shipImage = document.createElement("img");

shipImage.src = "ship.png";


// Asteroid image placeholder

const asteroidImage = document.createElement("img");

asteroidImage.src = "asteroid.png";


// Event listeners

document.addEventListener("keydown", handleKeyDown);

document.addEventListener("keyup", handleKeyUp);


// Key states

const keys = {};


// Start the game

requestAnimationFrame(gameLoop);


function gameLoop() {

    // Clear the canvas

    ctx.clearRect(0, 0, canvasWidth, canvasHeight);


    // Move and draw the ship

    moveShip();

    drawShip();


    // Move and draw the asteroids

    moveAsteroids();

    drawAsteroids();


    // Request the next frame

    requestAnimationFrame(gameLoop);

}


function moveShip() {

    if (keys["ArrowUp"]) {

        shipY -= 2;

    }

    if (keys["ArrowDown"]) {

        shipY += 2;

    }

    if (keys["ArrowLeft"]) {

        shipX -= 2;

    }

    if (keys["ArrowRight"]) {

        shipX += 2;

    }

}


function drawShip() {

    ctx.drawImage(shipImage, shipX - shipWidth / 2, shipY - shipHeight / 2, shipWidth, shipHeight);

}


function moveAsteroids() {

    for (let i = 0; i < asteroids.length; i++) {

        asteroids[i].x += asteroids[i].dx;

        asteroids[i].y += asteroids[i].dy;


        if (asteroids[i].x > canvasWidth) {

            asteroids[i].x = -asteroidWidth;

        } else if (asteroids[i].x < -asteroidWidth) {

            asteroids[i].x = canvasWidth;

        }


        if (asteroids[i].y > canvasHeight) {

            asteroids[i].y = -asteroidHeight;

        } else if (asteroids[i].y < -asteroidHeight) {

            asteroids[i].y = canvasHeight;

        }

    }

}


function drawAsteroids() {

    for (let i = 0; i < asteroids.length; i++) {

        ctx.drawImage(asteroidImage, asteroids[i].x, asteroids[i].y, asteroids[i].width, asteroids[i].height);

    }

}


function handleKeyDown(event) {

    keys[event.code] = true;

}


function handleKeyUp(event) {

    keys[event.code] = false;

}

</script>
